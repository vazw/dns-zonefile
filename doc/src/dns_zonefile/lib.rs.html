18/// DnsZonefile
19/// This is main struct 
20/// an interface to generate and parse function
21/// which will create Regex resource once using `DnsZonefile::default()`
22/// ```rust
23/// use dns_zonefile::{DnsZone, DnsZonefile};
24/// 
25/// fn main() {
26///     let dns_zonefile = DnsZonefile::default();
27///     let zonefile_example = r#"
28/// $ORIGIN MYDOMAIN.COM.
29/// $TTL 3600
30/// @    IN    SOA    NS1.NAMESERVER.NET.    HOSTMASTER.MYDOMAIN.COM.    (
31///             1406291485     ;serial
32///             3600     ;refresh
33///             600     ;retry
34///             604800     ;expire
35///             86400     ;minimum ttl
36/// )
37/// 
38/// @    NS    NS1.NAMESERVER.NET.
39/// @    NS    NS2.NAMESERVER.NET.
40/// 
41///     IN  A    2.2.2.2
42///     A    1.1.1.1
43/// @    A    127.0.0.1
44/// www    A    127.0.0.1
45/// mail    A    127.0.0.1
46///             A 1.2.3.4
47/// tst 300 IN A 101.228.10.127;this is a comment
48/// 
49/// @    AAAA    ::1
50/// mail    AAAA    2001:db8::1
51/// A  200 AAAA  2001:db8::1
52/// "#;
53/// 
54///     let Ok(zone_data) = dns_zonefile.parse(zonefile_example) else { return;};
55///     println!("{}", serde_json::to_string_pretty(&zone_data).unwrap());
56/// 
57///     let zonefile = dns_zonefile.generate(&zone_data, None);
58///     println!("{zonefile}");
59/// }
60/// ```
61impl DnsZonefile {
62    /// generate Zonefile string from DnsZone data struct like so
63    /// 
64    pub fn generate(&self, dns_zone: &DnsZone, template: Option<&str>) -> String {
65        generator::generate(&self.ctx.generator, dns_zone, template)
66    }
67    /// parse data from zonfile to DnsZone struct
68    ///
69    /// then can convert into json later using serde
70    ///
71    /// THIS function **return** error for empty zone data rather then all none!
72    pub fn parse(&self, data: &str) -> Result<DnsZone, String> {
73        parser::parse(&self.ctx.parser, &self.ctx.parser_soa, data)
74    }
75}
76
77impl Default for DnsZonefile {
78    fn default() -> Self {
79        let parser = Regex::new(r"\s+").unwrap();
80        let generator = Regex::new(r"\n{2,}").expect("valid pattern");
81        let parser_soa = Regex::new(r"(?is)(SOA\s+.*?\s*\([\s\S]*?\))").expect("valid pattern");
82        Self { ctx: ContextRegex { parser, parser_soa, generator } }
83    }
84}

190    #[test]
191    fn test_forward_idempotence() {
192        // 1. Start with the text file
193        let original_text = fs::read_to_string("tests/zonefile_forward.txt")
194            .expect("Failed to read zonefile_forward.txt");
195        let dns_zonefile = DnsZonefile::default();
196
197        // 2. Parse it to a JSON object (struct)
198        let json1 = dns_zonefile.parse(&original_text).expect("correct!");
199
200        // 3. Generate it back to text
201        let generated_text = dns_zonefile.generate(&json1, None);
202
203        // 4. Parse the generated text back to a JSON object (struct)
204        let json2 = dns_zonefile.parse(&generated_text).expect("json");
205
206        // 5. The two JSON objects should be deeply equal
207        assert_eq!(json1, json2, "Parsed objects should be equal after a generate/parse cycle.");
208    }
